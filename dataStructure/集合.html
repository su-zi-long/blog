# 集合

# 什么是集合？

集合是由一组无序且唯一（即不能重复）的项组成的。该数据结构使用了与有限集合相同的数学概念，但应用在计算机科学的数据结构中。

Javascript 中的 Set 就是一个集合，下面我们将用 Set 来实现集合运算的代码

# 集合运算

集合是数学中基础的概念，在计算机领域也非常重要。它在计算机科学中的主要应用之一是数据库，而数据库是大多数应用程序的根基。集合被用于查询的设计和处理。当我们创建一条从关系型数据库（Oracle、Microsoft SQL Server、MySQL等）中获取一个数据集合的查询语句时，使用的就是集合运算，并且数据库也会返回一个数据集合。当我们创建一条SQL查询命令时，可以指定是从表中获取全部数据还是获取其中的子集；也可以获取两张表共有的数据、只存在于一张表中的数据（不存在于另一张表中），或是存在于两张表内的数据（通过其他运算）。这些SQL领域的运算叫作联接，而SQL联接的基础就是集合运算。

# 并集

对于给定的两个集合，返回一个包含两个集合中所有元素的新集合。

const set1 = new Set([1, 2, 3, 4, 5]);
const set2 = new Set([3, 4, 5, 6, 7]);

function union(set1, set2) {
  return new Set([...set1, ...set2]);
}

const result = union(set1, set2);
console.log(result); // Set(7) { 1, 2, 3, 4, 5, 6, 7 }

# 交集

对于给定的两个集合，返回一个包含两个集合中共有元素的新集合。

const set1 = new Set([1, 2, 3, 4, 5]);
const set2 = new Set([3, 4, 5, 6, 7]);

function intersection(set1, set2) {
  const set = new Set();
  set1.forEach((item) => {
    if (set2.has(item)) set.add(item);
  });
  return set;
}

const result = intersection(set1, set2);
console.log(result); // Set(7) { 3, 4, 5 }

  { name: 'suzilong2' },

  { name: 'suzilong2' },

/**
* 多个数组的交集
* equals: 判断元素是否相等的方法
* twoDimensionalArray: 二维数组
*/
function arraysIntersection(equals, twoDimensionalArray) {
  if (twoDimensionalArray.length === 0) return [];
  if (twoDimensionalArray.length === 1) return twoDimensionalArray[0];

  let result = [];
  const intersection = (equals, array1, array2) => {
    const result = [];
    for (let i = 0; i < array1.length; i++) {
      let bool = false;
      for (let j = 0; j < array2.length; j++) {
        if (equals(array1[i], array2[j])) {
          bool = true;
          break;
        }
      }
      if (bool) result.push(array1[i]);
    }
    return result;
  };

  twoDimensionalArray.sort(function (a, b) {
    return a.length - b.length;
  });

  result = intersection(equals, twoDimensionalArray[0], twoDimensionalArray[1]);

  if (twoDimensionalArray.length > 2) {
    for (let i = 2; i < twoDimensionalArray.length; i++) {
      result = intersection(equals, result, twoDimensionalArray[i]);
    }
  }

  return result;
}

function equals(a, b) {
  return a.name === b.name;
}
const result = arraysIntersection(equals, [array3, array2, array1 ]);
console.log(result); // [{ name: 'suzilong2' }, { name: 'suzilong3' }]

# 差集

对于给定的两个集合，返回一个包含所有存在于第一个集合且不存在于第二个集合的元素的新集合。

const set1 = new Set([1, 2, 3, 4, 5]);
const set2 = new Set([3, 4, 5, 6, 7]);

function difference(set1, set2) {
  const set = new Set();
  set1.forEach((item) => {
    if (!set2.has(item)) set.add(item);
  });
  return set;
}

const result = difference(set1, set2);
console.log(result); // Set(7) { 1, 2 }

# 子集

验证一个给定集合是否是另一集合的子集。

const set1 = new Set([1, 2, 3, 4, 5]);
const set2 = new Set([2, 3, 4]);

function isSubset(parentSet, subset) {
  return [...subset].every(item => parentSet.has(item));
}

const result = isSubset(set1, set2);
console.log(result); // true

# 多重集

集合数据结构不允许存在重复的元素。但是，在数学中有一个叫作 多重集 的概念，允许我们向集合中插入重复元素。多重集（MultiSet）在计算集合中元素的出现次数时很有用。

class MultiSet {
  constructor() {
    this.data = new Map();
  }

  has(element) {
    return this.data.has(element);
  }

  set(element, count = 1) {
    this.data.set(element, this.count(element) + count)
  }

  setCount(element, count = 1) {
    if (count === 0) {
      return this.delete(element)
    }
    this.data.set(element, count)
  }

  count(element) {
    return this.data.get(element) || 0
  }

  delete(element) {
    return this.data.delete(element)
  }
}

const multiSet = new MultiSet()
multiSet.set(1, 2)
multiSet.setCount(1, 3)
multiSet.set(1)
multiSet.set(1)
console.log(multiSet.count(1)); // 5
