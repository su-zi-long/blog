# 链表

要存储多个元素，数组（或列表）可能是最常用的数据结构。正如本书之前提到的，每种语言都实现了数组。这种数据结构非常方便，提供了一个便利的[]语法来访问其元素。

然而，这种数据结构有一个缺点：（在大多数语言中）数组的大小是固定的，从数组的起点或中间插入或移除项的成本很高，因为需要移动元素。（尽管我们已经学过，JavaScript有来自Array类的方法可以帮我们做这些事，但背后的情况同样如此。）

链表存储有序的元素集合，但不同于数组，链表中的元素在内存中并不是连续放置的。每个元素由一个存储元素本身的节点和一个指向下一个元素的引用（也称指针或链接）组成。

相对于传统的数组，链表的一个好处在于，添加或移除元素的时候不需要移动其他元素。然而，链表需要使用指针，因此实现链表时需要额外注意。在数组中，我们可以直接访问任何位置的任何元素，而要想访问链表中间的一个元素，则需要从起点（表头）开始迭代链表直到找到所需的元素。

链表在现实生活中的例子：

  • 寻宝游戏。你有一条线索，这条线索就是指向寻找下一条线索的地点的指针。你顺着这条链接去下一个地点，得到另一条指向再下一处的线索。得到链表中间的线索的唯一办法，就是从起点（第一条线索）顺着链表寻找。

  • 火车。一列火车是由一系列车厢（也称车皮）组成的。每节车厢或车皮都相互连接。你很容易分离一节车皮，改变它的位置、添加或移除它。下图演示了一列火车。每节车皮都是链表的元素，车皮间的连接就是指针。

class LinkedList {
  constructor(equals) {
    this.head = null;
    this.length = 0;
    if (typeof equals === "function") this._equals = equals;
  }

  _equals(element1, element2) {
    return element1 === element2;
  }

  _generateNode(element, next = null) {
    return {
      element,
      next,
    };
  }

  push(element) {
    this.insert(element, this.length);
  }

  insert(element, index) {
    if (index < 0 || index > this.length) return false;
    if (index === 0) {
      this.head = this._generateNode(
        element,
        this.head === null ? null : this.head
      );
      this.length++;
      return true;
    }
    let node = this.head;
    for (let i = 1; i <= index; i++) {
      if (i === index) {
        node.next = this._generateNode(element, node.next);
        this.length++;
        return true;
      }
      node = node.next;
    }
  }

  indexOf(element) {
    let index = 0;
    let current = this.head;
    while (index < this.length) {
      if (this._equals(element, current.element)) return index;
      index++;
      current = current.next;
    }
    return -1;
  }

  remove(element) {
    let previousNode = null;
    let currentNode = this.head;
    for (let i = 0; i < this.length; i++) {
      if (this._equals(element, currentNode.element)) {
        if (previousNode) {
          previousNode.next = currentNode.next;
        } else {
          this.head = currentNode.next;
        }
        this.length--;
        return true;
      }
      previousNode = currentNode;
      currentNode = currentNode.next;
    }
  }

  removeAt(index) {
    if (index < 0 || index >= this.length) return false;
    if (index === 0) {
      this.head = this.head.next;
      return true;
    }
    let node = this.head;
    for (let i = 1; i < this.length; i++) {
      if (index === i) {
        node.next = node.next.next;
        this.length--;
        return true;
      }
      node = node.next;
    }
  }

  isEmpty() {
    return this.length === 0;
  }

  getLength() {
    return this.length;
  }
}

# 双向链表

链表有多种不同的类型，本节介绍双向链表。双向链表和普通链表的区别在于，在链表中，一个节点只有链向下一个节点的链接；而在双向链表中，链接是双向的：一个链向下一个元素，另一个链向前一个元素，如下图所示。

class doublyLinkedList {
  constructor(equals) {
    this.head = null;
    this.tail = null
    this.length = 0;
    if (typeof equals === "function") this._equals = equals;
  }

  _equals(element1, element2) {
    return element1 === element2;
  }

  _generateNode(element, prev = null, next = null) {
    return {
      prev,
      element,
      next,
    }
  }

  push(element) {
    this.insert(element, this.length)
  }

  insert(element, index) {
    if (index < 0 || index > this.length) return false
    if (index === 0) {
      if (this.head) {
        const node = this.head
        this.head = this._generateNode(element, null, node)
        node.prev = this.head
      } else {
        this.head = this.tail = this._generateNode(element, null, null)
      }
    } else if (index === this.length) {
      const tail = this.tail
      this.tail = this._generateNode(element, tail, null)
      tail.next = this.tail
    } else {
      let currentNode = this.head
      for (let i = 0; i < this.length; i++) {
        if (+i === +index) {
          const prevNode = currentNode.prev
          currentNode.prev = prevNode.next = this._generateNode(element, prevNode, currentNode)
          break
        }
        currentNode = currentNode.next
      }
    }
    this.length++;
    return true
  }

  indexOf(element) {
    let result = -1
    this.iteration((node, index) => {
      if (this._equals(element, node.element)) {
        result = index
        return false
      }
    })
    return result
  }

  remove(element) {
    let result = false
    this.iteration((node, index) => {
      if (!this._equals(element, node.element)) return
      const prevNode = node.prev
      if (index === 0) {
        this.head = node.next
        this.head.prev = null
      } else if (index === this.length - 1) {
        prevNode.next = null
        this.tail = prevNode
      } else {
        const nextNode = node.next
        prevNode.next = nextNode
        nextNode.prev = prevNode
      }
      this.length--
      result = true
      return false
    })
    return result
  }

  removeAt(index) {
    let result = false
    this.iteration((node, i) => {
      if (+index !== +i) return
      const prevNode = node.prev
      if (i === 0) {
        this.head = node.next
        this.head.prev = null
      } else if (i === this.length - 1) {
        prevNode.next = null
        this.tail = prevNode
      } else {
        const nextNode = node.next
        prevNode.next = nextNode
        nextNode.prev = prevNode
      }
      this.length--
      result = true
      return false
    })
    return result
  }

  isEmpty() {
    return this.length === 0
  }

  getLength() {
    return this.length
  }

  iteration(cb) {
    let node = this.head
    let i = 0
    while (node) {
      try {
        const result = cb(node, i)
        if (result === false) return
      } catch (error) {
        console.error(error)
      }
      node = node.next
      i++
    }
  }
}

# 循环链表

循环链表可以像链表一样只有单向引用，也可以像双向链表一样有双向引用。循环链表和链表之间唯一的区别在于，最后一个元素指向下一个元素的指针（tail.next）不是引用undefined，而是指向第一个元素（head）

双向循环链表有指向head元素的tail.next和指向tail元素的head.prev。
