# 求多个数组的交集

const array1 = [
  { name: 'suzilong1' },
  { name: 'suzilong2' },
  { name: 'suzilong3' },
];
const array2 = [
  { name: 'suzilong2' },
  { name: 'suzilong3' },
  { name: 'suzilong4' },
];
const array3 = [
  { name: 'suzilong2' },
  { name: 'suzilong3' },
  { name: 'suzilong4' },
  { name: 'suzilong5' },
];

function arraysIntersection(twoDimensionalArray, equals) {
  if (twoDimensionalArray.length <= 1) return [];
  if (typeof equals !== "function") {
    equals = (a, b) => a === b;
  }
  let result = [];
  const intersection = (equals, array1, array2) => {
    const result = [];
    for (let i = 0; i < array1.length; i++) {
      let bool = false;
      for (let j = 0; j < array2.length; j++) {
        if (equals(array1[i], array2[j])) {
          bool = true;
          break;
        }
      }
      if (bool) result.push(array1[i]);
    }
    return result;
  };

  twoDimensionalArray.sort(function (a, b) {
    return a.length - b.length;
  });

  result = intersection(equals, twoDimensionalArray[0], twoDimensionalArray[1]);

  if (twoDimensionalArray.length > 2) {
    for (let i = 2; i < twoDimensionalArray.length; i++) {
      result = intersection(equals, result, twoDimensionalArray[i]);
    }
  }

  return result;
}

function equals(a, b) {
  return a.name === b.name;
}

const result = arraysIntersection([array1, array2, array3], equals);
console.log(result); // [{ name: 'suzilong2' }, { name: 'suzilong3' }]
